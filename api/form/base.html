· 直接使用 ElPlusForm

vue
<ElPlusForm v-model="formData" v-bind="formConfig" />

· setup 中

ts
let formData = reactive({} as any)
const formConfig = ref({
  requestFn: (formData: any) => new Promise(() => {}),
  column: 2,
  formDesc: {
    goodsEas: { type: 'input', label: '商品名称', required: true },
    goodsSku: { type: 'input', label: '商品条码', attrs: { maxlength: 30 } },
    goodsName: { type: 'input', label: '商品主名称', required: true, attrs: { maxlength: 50 } },
    goodsSpec: { type: 'input', label: '规格', required: true, attrs: { maxlength: 50 } },
    goodsUnitType: { type: 'select', label: '最小零售单位', required: true, options: 'goodsUnitTypeList' },
    minUnit: { type: 'number', label: '箱规', required: true },
    taxRate: { type: 'select', label: '税率', default: 0.13, required: true, options: 'taxRateList' },
    expiration: { type: 'select', label: '保质期', required: true, options: 'expirationList', attrs: { filterable: true } },
    producer: { type: 'input', label: '产地' },
    productArea: { type: 'input', label: '产区' },
    imageList: { type: 'upload', label: '商品图片', multiple: true, colspan: 2 },
    remark: { type: 'textarea', label: '备注', colspan: 2 }
  } as IFormDesc
} as IFormConfig)
